// How to use venv for developer
// Virtual Environment
venv\Scripts\activate // ป้องกันปัญหา "เวอร์ชันตีกัน" (Dependency Hell)

python manage.py runserver



// How to step create database postgreSQL
docker-compose.yml
docker-compose up -d // docker-compose down (-v)

// โครงสร้าง db
python -m pip install Pillow // library ที่ทำให้สามารถอ่าน หรือตรวจสอบไฟล์รูปภาพด้วยได้
python manage.py makemigrations // ให้เข้าไปอ่าน models.py
python manage.py migrate // สั่งสร้างตาราง
python manage.py createsuperuser // สร้าง admin ไว้ login ทดสอบระบบ ในส่วน email ไม่ต้องใส่ก็ได้ข้ามไปเลย
python manage.py runserver // สั่งรัน api backend
